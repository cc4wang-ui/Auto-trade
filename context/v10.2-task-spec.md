## 四、輸出規格

### 變數定義（放在 v10.1 私人信貸子模組之後）

```pine
// ═══════════════════════════════════════════
// 不可控變數子模組（v10.2 新增）
// ═══════════════════════════════════════════

// ─── 1. 銀行體系健康度 ─────────────────────
// KRE 一月變化（D 週期 [22] = 約一個月）
float kre_1m_ago = nz(kre_close[22], kre_close)
float kre_1m_change = na(kre_close) or na(kre_1m_ago) or kre_1m_ago == 0 ? na : ((kre_close - kre_1m_ago) / kre_1m_ago) * 100.0

// BKX 一月變化（用於對照 — 區域 vs 大型銀行分歧）
float bkx_1m_ago = nz(bkx_close[22], bkx_close)
float bkx_1m_change = na(bkx_close) or na(bkx_1m_ago) or bkx_1m_ago == 0 ? na : ((bkx_close - bkx_1m_ago) / bkx_1m_ago) * 100.0

// 區域 vs 大型銀行分歧（>5% 是早期警訊）
float bank_divergence = na(kre_1m_change) or na(bkx_1m_change) ? na : math.abs(kre_1m_change - bkx_1m_change)

string bank_health_level = na(kre_1m_change) ? "N/A" :
                           kre_1m_change <= kreCrisisDrop ? "CRISIS" :
                           kre_1m_change <= kreWarnDrop  ? "WARNING" :
                           kre_1m_change < 0             ? "ELEVATED" :
                           "NORMAL"

// ─── 2. 通膨預期 ───────────────────────────
string inflation_exp_level = na(tips_be_10y) ? "N/A" :
                             tips_be_10y > tipsCrisisLevel or (not na(forward_5y5y) and forward_5y5y > fwd5y5yCrisisLevel) ? "CRISIS" :
                             tips_be_10y > tipsWarnLevel ? "WARNING" :
                             tips_be_10y > 2.3            ? "ELEVATED" :
                             "NORMAL"

// 通膨預期週變化（5 個 daily bar）
float tips_5d_ago = nz(tips_be_10y[5], tips_be_10y)
float tips_weekly_jump = na(tips_be_10y) or na(tips_5d_ago) ? na : tips_be_10y - tips_5d_ago
bool tips_acute_jump = not na(tips_weekly_jump) and tips_weekly_jump > 0.15

// ─── 3. VIX 期限結構 ───────────────────────
float vix_term_ratio = na(vix9d_close) or vix9d_close == 0 ? na : vix_close / vix9d_close

string vix_term_level = na(vix_term_ratio) ? "N/A" :
                        vix_term_ratio >= vixDeepBackwardThreshold ? "HALT" :
                        vix_term_ratio >= vixBackwardThreshold     ? "CRISIS" :
                        vix_term_ratio >= 0.95                      ? "WARNING" :
                        "NORMAL"

bool vix_term_inverted = not na(vix_term_ratio) and vix_term_ratio >= vixBackwardThreshold

// ─── 4. Fed 資產負債表變化（QT 速度）──────
// WALCL 是週度資料，[4] = 約一個月
float walcl_1m_ago = nz(walcl_total[4], walcl_total)
float walcl_1m_change = na(walcl_total) or na(walcl_1m_ago) or walcl_1m_ago == 0 ? na : ((walcl_total - walcl_1m_ago) / walcl_1m_ago) * 100.0

string fed_balance_level = na(walcl_1m_change) ? "N/A" :
                           walcl_1m_change <= walclQTSpeedCrisis ? "CRISIS" :
                           walcl_1m_change <= walclQTSpeedWarn   ? "WARNING" :
                           walcl_1m_change > 1.0                  ? "QE_MODE" :  // 大幅擴表 = Fed 救市
                           "NORMAL"

// ─── 5. 美元緊縮加速 ───────────────────────
float dxy_1m_ago = nz(dxy_close[22], dxy_close)
float dxy_1m_change = na(dxy_close) or na(dxy_1m_ago) or dxy_1m_ago == 0 ? na : ((dxy_close - dxy_1m_ago) / dxy_1m_ago) * 100.0

string dollar_squeeze_level = na(dxy_1m_change) ? "N/A" :
                              dxy_1m_change > dxyTightenSpeed     ? "WARNING" :  // 緊縮加速
                              dxy_1m_change < -dxyTightenSpeed    ? "ALERT" :    // 信任危機警訊
                              "NORMAL"

// ─── 6. 油價地緣 ───────────────────────────
string oil_geo_level = na(wti_close) ? "N/A" :
                       wti_close > wtiCrisisLevel ? "CRISIS" :
                       wti_close > wtiWarnLevel   ? "WARNING" :
                       wti_close < 60              ? "DEMAND_SHOCK" :  // 需求面警訊
                       "NORMAL"

// ═══ 整合：CRISIS 變數計數 ═══
int crisis_count = 0
if bank_health_level == "CRISIS"
    crisis_count := crisis_count + 1
if inflation_exp_level == "CRISIS"
    crisis_count := crisis_count + 1
if vix_term_level == "CRISIS" or vix_term_level == "HALT"
    crisis_count := crisis_count + 1
if fed_balance_level == "CRISIS"
    crisis_count := crisis_count + 1
if oil_geo_level == "CRISIS"
    crisis_count := crisis_count + 1
// dollar_squeeze 不計入 CRISIS（沒有 CRISIS level，只有 WARNING/ALERT）

// ═══ WARNING 變數計數 ═══
int warning_count = 0
if bank_health_level == "WARNING" or bank_health_level == "ELEVATED"
    warning_count := warning_count + 1
if inflation_exp_level == "WARNING" or inflation_exp_level == "ELEVATED"
    warning_count := warning_count + 1
if vix_term_level == "WARNING"
    warning_count := warning_count + 1
if fed_balance_level == "WARNING"
    warning_count := warning_count + 1
if dollar_squeeze_level == "WARNING" or dollar_squeeze_level == "ALERT"
    warning_count := warning_count + 1
if oil_geo_level == "WARNING"
    warning_count := warning_count + 1
```

### Regime Filter 整合（修改 v10.1 既有邏輯）

**v10.1 既有 regime 判定**（找到並擴充）：

```pine
// 原始 regime（基於同步性 + VIX）
string regime_base = vix_deep_backwardation ? "HALT" :
                     us2y_roc_extreme       ? "SHOCK" :
                     sync_index > sync_danger or vix_backwardation ? "CRISIS" :
                     sync_index > sync_warn or stability_low       ? "WARNING" :
                     "NORMAL"

// v10.1 信用壓力升級
string regime_after_hy = regime_base
if hy_force_crisis and regime_base != "HALT" and regime_base != "SHOCK"
    regime_after_hy := "CRISIS"
else if hy_force_warning and regime_base == "NORMAL"
    regime_after_hy := "WARNING"
```

**v10.2 新增邏輯**（接在後面，激進升級）：

```pine
// ═══ v10.2：不可控變數強制升級（激進邏輯）═══
string regime = regime_after_hy

// VIX 深度倒掛 → HALT（最高優先級）
if vix_term_level == "HALT"
    regime := "HALT"

// 任一 CRISIS 變數 → 強制 CRISIS（除非已 HALT/SHOCK）
if crisis_count >= 1 and regime != "HALT" and regime != "SHOCK"
    regime := "CRISIS"

// 急性事件強制升級
bool acute_uncontrol_event = tips_acute_jump or vix_term_inverted
if acute_uncontrol_event and regime == "NORMAL"
    regime := "WARNING"
else if acute_uncontrol_event and regime == "WARNING"
    regime := "CRISIS"

// 多 WARNING 累積 → 升級
if warning_count >= 3 and regime == "NORMAL"
    regime := "WARNING"
else if warning_count >= 4 and regime == "WARNING"
    regime := "CRISIS"

// 升級原因（給 dashboard 顯示）
string uncontrol_upgrade_reason = ""
if regime != regime_after_hy
    if vix_term_level == "HALT"
        uncontrol_upgrade_reason := "VIX 深度倒掛"
    else if crisis_count >= 1
        uncontrol_upgrade_reason := "CRISIS 變數 " + str.tostring(crisis_count) + " 個"
    else if acute_uncontrol_event
        uncontrol_upgrade_reason := tips_acute_jump ? "通膨預期跳升" : "VIX 倒掛"
    else if warning_count >= 3
        uncontrol_upgrade_reason := "WARNING 累積 " + str.tostring(warning_count) + " 個"
```

## 五、🔴 Pine Script 陷阱清單（必讀）

### 從 pattern-detector-pitfalls.md

- ✅ **P3-5**：條件鏈中 array.get / nested condition — 本次大量使用 string 比較
- ✅ **P3-2**：變數宣告順序 — 6 個新 request.security 必須在 module 開頭
- ✅ **P3-1**：ta.* 函數在 global scope — 本次無 ta.* 使用

### 從 dev-guide.md

- ✅ **Gotcha #5**：table 行數預留 — 新增 ~16 row，table.new rows=50 起跳
- ✅ **Gotcha #6**：NaN 防護 — 所有 [N] 索引必須 nz() 包
- ✅ **Gotcha #2**：strategy 不支援 alertcondition
- ✅ **Gotcha #27**（v10 新增）：request.security 容量管理 — 確認 13→19/40

### 本次新增的潛在陷阱

#### 陷阱 1：FRED:WALCL 是週度資料

- 60min 主圖上 [4] = 4 週 ≈ 一個月
- 但 lookahead_off 模式下，週末沒新資料時會 forward fill
- **驗證**：載入後手動確認 walcl_total 在週末的數值是否合理

#### 陷阱 2：AMEX:KRE 在某些帳戶找不到

- Cross 帳戶可能要用 NYSEARCA:KRE 或 NASDAQ:KRE
- **驗證**：載入後看 dashboard 區域銀行欄是否顯示 N/A
- 如果 N/A → 試 fallback ticker

#### 陷阱 3：crisis_count 累計順序

- 所有 string 比較必須在 crisis_count 計算之前完成
- Pine Script 是順序執行，順序錯就抓不到

#### 陷阱 4：regime 升級的優先順序

- HALT 最高（VIX 深度倒掛）
- 然後 SHOCK（2Y ROC）
- 然後 v10.1 HY CRISIS
- 然後 v10.2 不可控變數 CRISIS
- 不可降級

```pine
// 正確順序：
string regime = regime_after_hy  // 從 v10.1 接手
if vix_term_level == "HALT"
    regime := "HALT"  // 最高優先
else if crisis_count >= 1 and regime != "SHOCK"
    regime := "CRISIS"  // 不能降級已是 SHOCK 的
// ...
```

#### 陷阱 5：US Treasury / FRED 資料延遲

- TIPS BE / 5Y5Y / WALCL 都來自 FRED，可能延遲 1-2 天
- 必須 `na()` 包所有計算
- **驗證**：載入週末看是否仍能顯示前週數據
